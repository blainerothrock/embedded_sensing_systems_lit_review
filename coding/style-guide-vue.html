useWorkspaceStore — Composition API style

// stores/workspace.ts import { defineStore } from 'pinia' import { ref, computed } from 'vue' export const useWorkspaceStore = defineStore('workspace', () => { const papers = ref<Paper[]>([]) const activePaperId = ref<string | null>(null) const annotations = ref<Annotation[]>([]) const activePaper = computed(() => papers.value.find(p => p.id === activePaperId.value) ) const annotationsForActivePaper = computed(() => annotations.value.filter(a => a.paperId === activePaperId.value) ) function selectPaper(id: string) { activePaperId.value = id } function updateNote(annotationId: string, md: string) { const ann = annotations.value.find(a => a.id === annotationId) if (ann) ann.contentMd = md // db.run('UPDATE annotations SET content_md=? WHERE id=?', [md, id]) } return { papers, activePaperId, annotations, activePaper, annotationsForActivePaper, selectPaper, updateNote, } })

MarkdownNote.vue — Alpine → Vue migration

Before (Alpine)

<!-- local state, no store --> <div x-data="{ raw: '', preview: false, save() { // no way to persist! } }">

After (Vue + Pinia)

<!-- props in, store action out --> const props = defineProps<{ annotationId: string }>() const ws = useWorkspaceStore() const raw = ref(ws.getNote(props.annotationId)) const save = () => ws.updateNote(props.annotationId, raw.value)
